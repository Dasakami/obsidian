```dataviewjs
// завтрашняя дата
let tomorrow = moment().add(2, "days").format("DD.MM.YYYY");

// собираем все задачи в папке
let tasks = dv.pages('"Колледж/Пары"')
    .file.tasks
    .filter(t => t.text.includes(tomorrow));

let grouped = {};

// группируем по файлу (предмету)
for (let t of tasks) {
    let fileName = t.path;
    if (!grouped[fileName]) grouped[fileName] = [];
    grouped[fileName].push(t);
}

// формируем таблицу
let table = [];

for (let [fileName, tArr] of Object.entries(grouped)) {
    let total = tArr.length;
    let done = tArr.filter(t => t.completed).length;
    let progress = done / total;
    let status = progress >= 0.8 ? "Выполнено ✅" : "❌";
    table.push([dv.fileLink(fileName), tomorrow, status]);
}

dv.table(["Предмет", "Дата", "Статус"], table);
```
